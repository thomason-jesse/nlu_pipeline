163 static int32
164 phmm_link(allphone_search_t * allphs)
165 {
166  s3cipid_t ci, rc;
167  phmm_t *p, *p2;
168  int32 *rclist;
169  int32 i, n_link;
170  plink_t *l;
171  bin_mdef_t *mdef;
172  phmm_t **ci_phmm;
173 
174  mdef = ((ps_search_t *) allphs)->acmod->mdef;
175  ci_phmm = allphs->ci_phmm;
176 
177  rclist = (int32 *) ckd_calloc(mdef->n_ciphone + 1, sizeof(int32));
178 
179  /* Create successor links between PHMM nodes */
180  n_link = 0;
181  for (ci = 0; ci < mdef->n_ciphone; ci++) {
182  for (p = ci_phmm[ci]; p; p = p->next) {
183  /* Build rclist for p */
184  i = 0;
185  for (rc = 0; rc < mdef->n_ciphone; rc++) {
186  if (bitvec_is_set(p->rc, rc))
187  rclist[i++] = rc;
188  }
189  rclist[i] = BAD_S3CIPID;
190 
191  /* For each rc in rclist, transition to PHMMs for rc if left context = ci */
192  for (i = 0; IS_S3CIPID(rclist[i]); i++) {
193  for (p2 = ci_phmm[rclist[i]]; p2; p2 = p2->next) {
194  if (bitvec_is_set(p2->lc, ci)) {
195  /* transition from p to p2 */
196  l = (plink_t *) ckd_calloc(1, sizeof(*l));
197  l->phmm = p2;
198  l->next = p->succlist;
199  p->succlist = l;
200 
201  n_link++;
202  }
203  }
204  }
205  }
206  }
207 
208  ckd_free(rclist);
209 
210  return n_link;
211 }

216 static int
217 phmm_build(allphone_search_t * allphs)
218 {
219  phmm_t *p, **pid2phmm;
220  bin_mdef_t *mdef;
221  int32 lrc_size;
222  uint32 *lc, *rc;
223  s3pid_t pid;
224  s3cipid_t ci;
225  s3cipid_t *filler;
226  int n_phmm, n_link;
227  int i, nphone;
228 
229  mdef = ((ps_search_t *) allphs)->acmod->mdef;
230  allphs->ci_phmm =
231  (phmm_t **) ckd_calloc(bin_mdef_n_ciphone(mdef), sizeof(phmm_t *));
232  pid2phmm =
233  (phmm_t **) ckd_calloc(bin_mdef_n_phone(mdef), sizeof(phmm_t *));
234 
235  /* For each unique ciphone/triphone entry in mdef, create a PHMM node */
236  n_phmm = 0;
237  nphone = allphs->ci_only ? bin_mdef_n_ciphone(mdef) : bin_mdef_n_phone(mdef);
238  E_INFO("Building PHMM net of %d phones\n", nphone);
239  for (pid = 0; pid < nphone; pid++) {
240  if ((p = phmm_lookup(allphs, pid)) == NULL) {
241  //not found, should be created
242  p = (phmm_t *) ckd_calloc(1, sizeof(*p));
243  hmm_init(allphs->hmmctx, &(p->hmm), FALSE,
244  mdef_pid2ssid(mdef, pid), mdef->phone[pid].tmat);
245  p->pid = pid;
246  p->ci = bin_mdef_pid2ci(mdef, pid);
247  p->succlist = NULL;
248  p->next = allphs->ci_phmm[p->ci];
249  allphs->ci_phmm[p->ci] = p;
250  n_phmm++;
251  }
252  pid2phmm[pid] = p;
253  }
254 
255  /* Fill out bitvecs of each PHMM node, alloc continuous memory chunk for context bitvectors */
256  lrc_size = bitvec_size(bin_mdef_n_ciphone(mdef));
257  lc = ckd_calloc(n_phmm * 2 * lrc_size, sizeof(bitvec_t));
258  rc = lc + (n_phmm * lrc_size);
259  for (ci = 0; ci < mdef->n_ciphone; ci++) {
260  for (p = allphs->ci_phmm[ci]; p; p = p->next) {
261  p->lc = lc;
262  lc += lrc_size;
263  p->rc = rc;
264  rc += lrc_size;
265  }
266  }
267 
268  /* Fill out lc and rc bitmaps (remember to map all fillers to each other!!) */
269  filler =
270  (s3cipid_t *) ckd_calloc(bin_mdef_n_ciphone(mdef) + 1,
271  sizeof(s3cipid_t));
272 
273  /* Connect fillers */
274  i = 0;
275  for (ci = 0; ci < bin_mdef_n_ciphone(mdef); ci++) {
276  p = pid2phmm[ci];
277  bitvec_set_all(p->lc, bin_mdef_n_ciphone(mdef));
278  bitvec_set_all(p->rc, bin_mdef_n_ciphone(mdef));
279  if (mdef->phone[ci].info.ci.filler) {
280  filler[i++] = ci;
281  }
282  }
283  filler[i] = BAD_S3CIPID;
284 
285 
286  /* Loop over cdphones only if ci_only is not set */
287  for (pid = bin_mdef_n_ciphone(mdef); pid < nphone;
288  pid++) {
289  p = pid2phmm[pid];
290 
291  if (mdef->phone[mdef->phone[pid].info.cd.ctx[1]].info.ci.filler) {
292  for (i = 0; IS_S3CIPID(filler[i]); i++)
293  bitvec_set(p->lc, filler[i]);
294  }
295  else
296  bitvec_set(p->lc, mdef->phone[pid].info.cd.ctx[1]);
297 
298  if (mdef->phone[mdef->phone[pid].info.cd.ctx[2]].info.ci.filler) {
299  for (i = 0; IS_S3CIPID(filler[i]); i++)
300  bitvec_set(p->rc, filler[i]);
301  }
302  else
303  bitvec_set(p->rc, mdef->phone[pid].info.cd.ctx[2]);
304  }
305  ckd_free(pid2phmm);
306  ckd_free(filler);
307 
308  /* Create links between PHMM nodes */
309  n_link = phmm_link(allphs);
310 
311  E_INFO("%d nodes, %d links\n", n_phmm, n_link);
312  return 0;
313 }

315 static void
316 phmm_free(allphone_search_t * allphs)
317 {
318  s3cipid_t ci;
319  bin_mdef_t *mdef;
320 
321  if (!allphs->ci_phmm)
322  //nothing to free
323  return;
324  ckd_free(allphs->ci_phmm[0]->lc);
325  mdef = ((ps_search_t *) allphs)->acmod->mdef;
326  for (ci = 0; ci < mdef_n_ciphone(mdef); ++ci) {
327  phmm_t *p, *next;
328 
329  for (p = allphs->ci_phmm[ci]; p; p = next) {
330  plink_t *l, *lnext;
331 
332  next = p->next;
333  for (l = p->succlist; l; l = lnext) {
334  lnext = l->next;
335  ckd_free(l);
336  }
337  hmm_deinit(&(p->hmm));
338  ckd_free(p);
339  }
340  }
341  ckd_free(allphs->ci_phmm);
342 }

345 static int32
346 phmm_eval_all(allphone_search_t * allphs, const int16 * senscr)
347 {
348  s3cipid_t ci;
349  phmm_t *p;
350  int32 best;
351  bin_mdef_t *mdef;
352  phmm_t **ci_phmm;
353 
354  mdef = ((ps_search_t *) allphs)->acmod->mdef;
355  ci_phmm = allphs->ci_phmm;
356 
357  best = WORST_SCORE;
358 
359  hmm_context_set_senscore(allphs->hmmctx, senscr);
360  for (ci = 0; ci < mdef->n_ciphone; ci++) {
361  for (p = ci_phmm[(unsigned) ci]; p; p = p->next) {
362  if (hmm_frame(&(p->hmm)) == allphs->frame) {
363  int32 score;
364  allphs->n_hmm_eval++;
365  score = hmm_vit_eval((hmm_t *) p);
366  if (score > best)
367  best = score;
368  }
369  }
370  }
371 
372  return best;
373 }

375 static void
376 phmm_exit(allphone_search_t * allphs, int32 best)
377 {
378  s3cipid_t ci;
379  phmm_t *p;
380  int32 th, nf;
381  history_t *h;
382  blkarray_list_t *history;
383  bin_mdef_t *mdef;
384  int32 curfrm;
385  phmm_t **ci_phmm;
386  int32 *ci2lmwid;
387 
388  th = best + allphs->pbeam;
389 
390  history = allphs->history;
391  mdef = ps_search_acmod(allphs)->mdef;
392  curfrm = allphs->frame;
393  ci_phmm = allphs->ci_phmm;
394  ci2lmwid = allphs->ci2lmwid;
395 
396  nf = curfrm + 1;
397 
398  for (ci = 0; ci < mdef->n_ciphone; ci++) {
399  for (p = ci_phmm[(unsigned) ci]; p; p = p->next) {
400  if (hmm_frame(&(p->hmm)) == curfrm) {
401 
402  if (hmm_bestscore(&(p->hmm)) >= th) {
403 
404  h = (history_t *) ckd_calloc(1, sizeof(*h));
405  h->ef = curfrm;
406  h->phmm = p;
407  h->hist = hmm_out_history(&(p->hmm));
408  h->score = hmm_out_score(&(p->hmm));
409 
410  if (!allphs->lm) {
411  h->tscore = allphs->inspen;
412  }
413  else {
414  if (h->hist > 0) {
415  int32 n_used;
416  history_t *pred =
417  blkarray_list_get(history, h->hist);
418 
419  if (pred->hist > 0) {
420  history_t *pred_pred =
421  blkarray_list_get(history,
422  h->hist);
423  h->tscore =
424  ngram_tg_score(allphs->lm,
425  ci2lmwid
426  [pred_pred->phmm->ci],
427  ci2lmwid[pred->
428  phmm->ci],
429  ci2lmwid[p->ci],
430  &n_used) >>
431  SENSCR_SHIFT;
432  }
433  else {
434  h->tscore =
435  ngram_bg_score(allphs->lm,
436  ci2lmwid
437  [pred->phmm->ci],
438  ci2lmwid[p->ci],
439  &n_used) >>
440  SENSCR_SHIFT;
441  }
442  }
443  else {
444  /*
445  * This is the beginning SIL and in srch_allphone_begin()
446  * it's inscore is set to 0.
447  */
448  h->tscore = 0;
449  }
450  }
451 
452  blkarray_list_append(history, h);
453 
454  /* Mark PHMM active in next frame */
455  hmm_frame(&(p->hmm)) = nf;
456  }
457  else {
458  /* Reset state scores */
459  hmm_clear(&(p->hmm));
460  }
461  }
462  }
463  }
464 }

466 static void
467 phmm_trans(allphone_search_t * allphs, int32 best,
468  int32 frame_history_start)
469 {
470  history_t *h;
471  phmm_t *from, *to;
472  plink_t *l;
473  int32 newscore, nf, curfrm;
474  int32 *ci2lmwid;
475  int32 hist_idx;
476 
477  curfrm = allphs->frame;
478  nf = curfrm + 1;
479  ci2lmwid = allphs->ci2lmwid;
480 
481  /* Transition from exited nodes to initial states of HMMs */
482  for (hist_idx = frame_history_start;
483  hist_idx < blkarray_list_n_valid(allphs->history); hist_idx++) {
484  h = blkarray_list_get(allphs->history, hist_idx);
485  from = h->phmm;
486  for (l = from->succlist; l; l = l->next) {
487  int32 tscore;
488  to = l->phmm;
489 
490  /* No LM, just use uniform (insertion penalty). */
491  if (!allphs->lm)
492  tscore = allphs->inspen;
493  else {
494  int32 n_used;
495  if (h->hist > 0) {
496  history_t *pred =
497  blkarray_list_get(allphs->history, h->hist);
498  tscore =
499  ngram_tg_score(allphs->lm,
500  ci2lmwid[pred->phmm->ci],
501  ci2lmwid[from->ci],
502  ci2lmwid[to->ci],
503  &n_used) >> SENSCR_SHIFT;
504  }
505  else {
506  tscore = ngram_bg_score(allphs->lm,
507  ci2lmwid[from->ci],
508  ci2lmwid[to->ci],
509  &n_used) >> SENSCR_SHIFT;
510  }
511  }
512 
513  newscore = h->score + tscore;
514  if ((newscore > best + allphs->beam)
515  && (newscore > hmm_in_score(&(to->hmm)))) {
516  hmm_enter(&(to->hmm), newscore, hist_idx, nf);
517  }
518  }
519  }
520 }

522 ps_search_t *
523 allphone_search_init(const char *name,
524  ngram_model_t * lm,
525  cmd_ln_t * config,
526  acmod_t * acmod, dict_t * dict, dict2pid_t * d2p)
527 {
528  int i;
529  bin_mdef_t *mdef;
530  allphone_search_t *allphs;
531 
532  allphs = (allphone_search_t *) ckd_calloc(1, sizeof(*allphs));
533  ps_search_init(ps_search_base(allphs), &allphone_funcs, PS_SEARCH_TYPE_ALLPHONE, name, config, acmod,
534  dict, d2p);
535  mdef = acmod->mdef;
536 
537  allphs->hmmctx = hmm_context_init(bin_mdef_n_emit_state(mdef),
538  acmod->tmat->tp, NULL, mdef->sseq);
539  if (allphs->hmmctx == NULL) {
540  ps_search_free(ps_search_base(allphs));
541  return NULL;
542  }
543 
544  allphs->ci_only = cmd_ln_boolean_r(config, "-allphone_ci");
545  allphs->lw = cmd_ln_float32_r(config, "-lw");
546 
547  phmm_build(allphs);
548 
549  if (lm) {
550  int32 silwid;
551 
552  allphs->lm = ngram_model_retain(lm);
553 
554  silwid = ngram_wid(allphs->lm, bin_mdef_ciphone_str(mdef,
555  mdef_silphone
556  (mdef)));
557  if (silwid == ngram_unknown_wid(allphs->lm)) {
558  E_ERROR("Phonetic LM does not have SIL phone in vocabulary\n");
559  allphone_search_free((ps_search_t *) allphs);
560  return NULL;
561  }
562 
563  allphs->ci2lmwid =
564  (int32 *) ckd_calloc(mdef->n_ciphone,
565  sizeof(*allphs->ci2lmwid));
566  for (i = 0; i < mdef->n_ciphone; i++) {
567  allphs->ci2lmwid[i] =
568  ngram_wid(allphs->lm,
569  (char *) bin_mdef_ciphone_str(mdef, i));
570  /* Map filler phones and other missing phones to silence if not found */
571  if (allphs->ci2lmwid[i] == ngram_unknown_wid(allphs->lm))
572  allphs->ci2lmwid[i] = silwid;
573  }
574  }
575  else {
576  E_WARN
577  ("Failed to load language model specified in -allphone, doing unconstrained phone-loop decoding\n");
578  allphs->inspen =
579  (int32) (logmath_log
580  (acmod->lmath, cmd_ln_float32_r(config, "-pip"))
581  * allphs->lw) >> SENSCR_SHIFT;
582  }
583 
584  allphs->n_tot_frame = 0;
585  allphs->frame = -1;
586  allphs->segments = NULL;
587 
588  /* Get search pruning parameters */
589  allphs->beam
590  =
591  (int32) logmath_log(acmod->lmath,
592  cmd_ln_float64_r(config, "-beam"))
593  >> SENSCR_SHIFT;
594  allphs->pbeam
595  =
596  (int32) logmath_log(acmod->lmath,
597  cmd_ln_float64_r(config, "-pbeam"))
598  >> SENSCR_SHIFT;
599 
600  /* LM related weights/penalties */
601  allphs->history = blkarray_list_init();
602 
603  /* Acoustic score scale for posterior probabilities. */
604  allphs->ascale = 1.0 / cmd_ln_float32_r(config, "-ascale");
605 
606  E_INFO("Allphone(beam: %d, pbeam: %d)\n", allphs->beam, allphs->pbeam);
607 
608  ptmr_init(&allphs->perf);
609 
610  return (ps_search_t *) allphs;
611 }

636 void
637 allphone_search_free(ps_search_t * search)
638 {
639  allphone_search_t *allphs = (allphone_search_t *) search;
640 
641 
642  double n_speech = (double)allphs->n_tot_frame
643  / cmd_ln_int32_r(ps_search_config(allphs), "-frate");
644 
645  E_INFO("TOTAL allphone %.2f CPU %.3f xRT\n",

698  hmm_enter(&(p->hmm), 0, 0, allphs->frame);

771 static void
772 allphone_backtrace(allphone_search_t * allphs, int32 f, int32 *out_score)
773 {
774  int32 best, hist_idx, best_idx;
775  int32 frm, last_frm;
776  history_t *h;
777  phseg_t *s;
778 
779  /* Clear old list */
780  allphone_clear_segments(allphs);
781 
782  frm = last_frm = f;
783  /* Find the first history entry for the requested frame */
784  hist_idx = blkarray_list_n_valid(allphs->history) - 1;
785  while (hist_idx > 0) {
786  h = blkarray_list_get(allphs->history, hist_idx);
787  if (h->ef <= f) {
788  frm = last_frm = h->ef;
789  break;
790  }
791  hist_idx--;
792  }
793 
794  if (hist_idx < 0)
795  return;
796 
797  /* Find bestscore */
798  best = (int32) 0x80000000;
799  best_idx = -1;
800  while (frm == last_frm && hist_idx > 0) {
801  h = blkarray_list_get(allphs->history, hist_idx);
802  frm = h->ef;
803  if (h->score > best && frm == last_frm) {
804  best = h->score;
805  best_idx = hist_idx;
806  }
807  hist_idx--;
808  }
809 
810  if (best_idx < 0)
811  return;
812 
813  if (out_score)
814  *out_score = best;
815 
816  /* Backtrace */
817  while (best_idx > 0) {
818  h = blkarray_list_get(allphs->history, best_idx);
819  s = (phseg_t *) ckd_calloc(1, sizeof(phseg_t));
820  s->ci = h->phmm->ci;
821  s->sf =
822  (h->hist >
823  0) ? ((history_t *) blkarray_list_get(allphs->history,
824  h->hist))->ef + 1 : 0;
825  s->ef = h->ef;
826  s->score = ascore(allphs, h);
827  s->tscore = h->tscore;
828  allphs->segments = glist_add_ptr(allphs->segments, s);
829 
830  best_idx = h->hist;
831  }
832 
833  return;
834 }
