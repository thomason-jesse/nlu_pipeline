98 ps_expand_file_config(ps_decoder_t *ps, const char *arg, const char *extra_arg,
99  const char *hmmdir, const char *file)
100 {
101  const char *val;
102  if ((val = cmd_ln_str_r(ps->config, arg)) != NULL) {
103  cmd_ln_set_str_extra_r(ps->config, extra_arg, val);
104  } else if (hmmdir == NULL) {
105  cmd_ln_set_str_extra_r(ps->config, extra_arg, NULL);

107  char *tmp = string_join(hmmdir, "/", file, NULL);
108  if (file_exists(tmp))

110  else
111  cmd_ln_set_str_extra_r(ps->config, extra_arg, NULL);
112  ckd_free(tmp);
113  }
114 }

123 static void
124 ps_expand_model_config(ps_decoder_t *ps)
125 {
126  char const *hmmdir, *featparams;
127 
128  /* Disable memory mapping on Blackfin (FIXME: should be uClinux in general). */
129 #ifdef __ADSPBLACKFIN__
130  E_INFO("Will not use mmap() on uClinux/Blackfin.");
131  cmd_ln_set_boolean_r(ps->config, "-mmap", FALSE);
132 #endif
133 
134  /* Get acoustic model filenames and add them to the command-line */
135  hmmdir = cmd_ln_str_r(ps->config, "-hmm");
136  ps_expand_file_config(ps, "-mdef", "_mdef", hmmdir, "mdef");
137  ps_expand_file_config(ps, "-mean", "_mean", hmmdir, "means");
138  ps_expand_file_config(ps, "-var", "_var", hmmdir, "variances");
139  ps_expand_file_config(ps, "-tmat", "_tmat", hmmdir, "transition_matrices");
140  ps_expand_file_config(ps, "-mixw", "_mixw", hmmdir, "mixture_weights");
141  ps_expand_file_config(ps, "-sendump", "_sendump", hmmdir, "sendump");
142  ps_expand_file_config(ps, "-fdict", "_fdict", hmmdir, "noisedict");
143  ps_expand_file_config(ps, "-lda", "_lda", hmmdir, "feature_transform");
144  ps_expand_file_config(ps, "-featparams", "_featparams", hmmdir, "feat.params");
145  ps_expand_file_config(ps, "-senmgau", "_senmgau", hmmdir, "senmgau");
146 
147  /* Look for feat.params in acoustic model dir. */
148  if ((featparams = cmd_ln_str_r(ps->config, "_featparams"))) {
149  if (NULL !=
150  cmd_ln_parse_file_r(ps->config, feat_defn, featparams, FALSE))
151  E_INFO("Parsed model-specific feature parameters from %s\n",
152  featparams);
153  }
154 
155  /* Print here because acmod_init might load feat.params file */
156  if (err_get_logfp() != NULL) {
157  cmd_ln_print_values_r(ps->config, err_get_logfp(), ps_args());
158  }

186 void

190  /* Set default acoustic and language models. */

192  if (hmmdir == NULL && hmmdir_exists(MODELDIR "/en-us/en-us")) {
193  hmmdir = MODELDIR "/en-us/en-us";
194  cmd_ln_set_str_r(config, "-hmm", hmmdir);
195  }
196 
197  const char *lmfile = cmd_ln_str_r(config, "-lm");
198 
199  if (lmfile == NULL && !cmd_ln_str_r(config, "-fsg")
200  && !cmd_ln_str_r(config, "-jsgf")
201  && !cmd_ln_str_r(config, "-lmctl")
202  && !cmd_ln_str_r(config, "-kws")
203  && !cmd_ln_str_r(config, "-keyphrase")
204  && file_exists(MODELDIR "/en-us/en-us.lm.bin")) {
205  lmfile = MODELDIR "/en-us/en-us.lm.bin";
206  cmd_ln_set_str_r(config, "-lm", lmfile);
207  }
208 
209  const char *dictfile = cmd_ln_str_r(config, "-dict");
210  if (dictfile == NULL && file_exists(MODELDIR "/en-us/cmudict-en-us.dict")) {
211  dictfile = MODELDIR "/en-us/cmudict-en-us.dict";
212  cmd_ln_set_str_r(config, "-dict", dictfile);
213  }
214 
215  /* Expand acoustic and language model filenames relative to installation
216  * path. */
217  if (hmmdir && !path_is_absolute(hmmdir) && !hmmdir_exists(hmmdir)) {
218  char *tmphmm = string_join(MODELDIR "/hmm/", hmmdir, NULL);
219  if (hmmdir_exists(tmphmm)) {
220  cmd_ln_set_str_r(config, "-hmm", tmphmm);
221  } else {
222  E_ERROR("Failed to find mdef file inside the model folder "
223  "specified with -hmm `%s'\n", hmmdir);
224  }
225  ckd_free(tmphmm);
226  }
227  if (lmfile && !path_is_absolute(lmfile) && !file_exists(lmfile)) {
228  char *tmplm = string_join(MODELDIR "/lm/", lmfile, NULL);
229  cmd_ln_set_str_r(config, "-lm", tmplm);
230  ckd_free(tmplm);
231  }
232  if (dictfile && !path_is_absolute(dictfile) && !file_exists(dictfile)) {
233  char *tmpdict = string_join(MODELDIR "/lm/", dictfile, NULL);
234  cmd_ln_set_str_r(config, "-dict", tmpdict);
235  ckd_free(tmpdict);
236  }
237 #endif

240 int
241 ps_reinit(ps_decoder_t *ps, cmd_ln_t *config)
242 {
243  const char *path;
244  const char *keyphrase;
245  int32 lw;
246 
247  if (config && config != ps->config) {
248  cmd_ln_free_r(ps->config);
249  ps->config = cmd_ln_retain(config);
250  }
251 
252  err_set_debug_level(cmd_ln_int32_r(ps->config, "-debug"));
253  /* Set up logging. We need to do this earlier because we want to dump
254  * the information to the configured log, not to the stderr. */
255  if (config && cmd_ln_str_r(ps->config, "-logfn")) {
256  if (err_set_logfile(cmd_ln_str_r(ps->config, "-logfn")) < 0) {
257  E_ERROR("Cannot redirect log output\n");
258  return -1;
259  }
260  }
261 
262  ps->mfclogdir = cmd_ln_str_r(ps->config, "-mfclogdir");
263  ps->rawlogdir = cmd_ln_str_r(ps->config, "-rawlogdir");
264  ps->senlogdir = cmd_ln_str_r(ps->config, "-senlogdir");
265 
266  /* Fill in some default arguments. */
267  ps_expand_model_config(ps);
268 
269  /* Free old searches (do this before other reinit) */
270  ps_free_searches(ps);
271  ps->searches = hash_table_new(3, HASH_CASE_YES);
272 
273  /* Free old acmod. */
274  acmod_free(ps->acmod);
275  ps->acmod = NULL;
276 
277  /* Free old dictionary (must be done after the two things above) */
278  dict_free(ps->dict);
279  ps->dict = NULL;
280 
281  /* Free d2p */
282  dict2pid_free(ps->d2p);
283  ps->d2p = NULL;
284 
285  /* Logmath computation (used in acmod and search) */
286  if (ps->lmath == NULL
287  || (logmath_get_base(ps->lmath) !=
288  (float64)cmd_ln_float32_r(ps->config, "-logbase"))) {
289  if (ps->lmath)
290  logmath_free(ps->lmath);
291  ps->lmath = logmath_init
292  ((float64)cmd_ln_float32_r(ps->config, "-logbase"), 0,
293  cmd_ln_boolean_r(ps->config, "-bestpath"));
294  }
295 
296  /* Acoustic model (this is basically everything that
297  * uttproc.c, senscr.c, and others used to do) */
298  if ((ps->acmod = acmod_init(ps->config, ps->lmath, NULL, NULL)) == NULL)
299  return -1;
300 
301 
302 
303  if (cmd_ln_int32_r(ps->config, "-pl_window") > 0) {
304  /* Initialize an auxiliary phone loop search, which will run in
305  * "parallel" with FSG or N-Gram search. */
306  if ((ps->phone_loop =
307  phone_loop_search_init(ps->config, ps->acmod, ps->dict)) == NULL)
308  return -1;
309  hash_table_enter(ps->searches,
310  ps_search_name(ps->phone_loop),
311  ps->phone_loop);
312  }
313 
314  /* Dictionary and triphone mappings (depends on acmod). */
315  /* FIXME: pass config, change arguments, implement LTS, etc. */
316  if ((ps->dict = dict_init(ps->config, ps->acmod->mdef)) == NULL)
317  return -1;
318  if ((ps->d2p = dict2pid_build(ps->acmod->mdef, ps->dict)) == NULL)
319  return -1;
320 
321  lw = cmd_ln_float32_r(ps->config, "-lw");
322 
323  /* Determine whether we are starting out in FSG or N-Gram search mode.
324  * If neither is used skip search initialization. */
325 
326  /* Load KWS if one was specified in config */
327  if ((keyphrase = cmd_ln_str_r(ps->config, "-keyphrase"))) {
328  if (ps_set_keyphrase(ps, PS_DEFAULT_SEARCH, keyphrase))
329  return -1;
330  ps_set_search(ps, PS_DEFAULT_SEARCH);
331  }
332 
333  if ((path = cmd_ln_str_r(ps->config, "-kws"))) {
334  if (ps_set_kws(ps, PS_DEFAULT_SEARCH, path))
335  return -1;
336  ps_set_search(ps, PS_DEFAULT_SEARCH);
337  }
338 
339  /* Load an FSG if one was specified in config */
340  if ((path = cmd_ln_str_r(ps->config, "-fsg"))) {
341  fsg_model_t *fsg = fsg_model_readfile(path, ps->lmath, lw);
342  if (!fsg)
343  return -1;
344  if (ps_set_fsg(ps, PS_DEFAULT_SEARCH, fsg)) {
345  fsg_model_free(fsg);
346  return -1;
347  }
348  fsg_model_free(fsg);
349  ps_set_search(ps, PS_DEFAULT_SEARCH);
350  }
351 
352  /* Or load a JSGF grammar */
353  if ((path = cmd_ln_str_r(ps->config, "-jsgf"))) {
354  if (ps_set_jsgf_file(ps, PS_DEFAULT_SEARCH, path)
355  || ps_set_search(ps, PS_DEFAULT_SEARCH))
356  return -1;
357  }
358 
359  if ((path = cmd_ln_str_r(ps->config, "-allphone"))) {
360  if (ps_set_allphone_file(ps, PS_DEFAULT_SEARCH, path)
361  || ps_set_search(ps, PS_DEFAULT_SEARCH))
362  return -1;
363  }
364 
365  if ((path = cmd_ln_str_r(ps->config, "-lm")) &&
366  !cmd_ln_boolean_r(ps->config, "-allphone")) {
367  if (ps_set_lm_file(ps, PS_DEFAULT_SEARCH, path)
368  || ps_set_search(ps, PS_DEFAULT_SEARCH))
369  return -1;
370  }
371 
372  if ((path = cmd_ln_str_r(ps->config, "-lmctl"))) {
373  const char *name;
374  ngram_model_t *lmset;
375  ngram_model_set_iter_t *lmset_it;
376 
377  if (!(lmset = ngram_model_set_read(ps->config, path, ps->lmath))) {
378  E_ERROR("Failed to read language model control file: %s\n", path);
379  return -1;
380  }
381 
382  for(lmset_it = ngram_model_set_iter(lmset);
383  lmset_it; lmset_it = ngram_model_set_iter_next(lmset_it)) {
384  ngram_model_t *lm = ngram_model_set_iter_model(lmset_it, &name);
385  E_INFO("adding search %s\n", name);
386  if (ps_set_lm(ps, name, lm)) {
387  ngram_model_set_iter_free(lmset_it);
388  ngram_model_free(lmset);
389  return -1;
390  }
391  }
392  ngram_model_free(lmset);
393 
394  name = cmd_ln_str_r(ps->config, "-lmname");
395  if (name)
396  ps_set_search(ps, name);
397  else {
398  E_ERROR("No default LM name (-lmname) for `-lmctl'\n");
399  return -1;
400  }
401  }
402 
403  /* Initialize performance timer. */
404  ps->perf.name = "decode";
405  ptmr_init(&ps->perf);
406 
407  return 0;
408 }

507  } else {

511  return 0;

906 ps_decode_raw(ps_decoder_t *ps, FILE *rawfh,
907  long maxsamps)
908 {
909  int16 *data;
910  long total, pos, endpos;
911 
912  ps_start_stream(ps);
913  ps_start_utt(ps);
914 
915  /* If this file is seekable or maxsamps is specified, then decode
916  * the whole thing at once. */
917  if (maxsamps != -1) {
918  data = ckd_calloc(maxsamps, sizeof(*data));
919  total = fread(data, sizeof(*data), maxsamps, rawfh);
920  ps_process_raw(ps, data, total, FALSE, TRUE);
921  ckd_free(data);
922  } else if ((pos = ftell(rawfh)) >= 0) {
923  fseek(rawfh, 0, SEEK_END);
924  endpos = ftell(rawfh);
925  fseek(rawfh, pos, SEEK_SET);
926  maxsamps = endpos - pos;
927 
928  data = ckd_calloc(maxsamps, sizeof(*data));
929  total = fread(data, sizeof(*data), maxsamps, rawfh);
930  ps_process_raw(ps, data, total, FALSE, TRUE);
931  ckd_free(data);
932  } else {
933  /* Otherwise decode it in a stream. */
934  total = 0;
935  while (!feof(rawfh)) {
936  int16 data[256];
937  size_t nread;
938 
939  nread = fread(data, sizeof(*data), sizeof(data)/sizeof(*data), rawfh);
940  ps_process_raw(ps, data, nread, FALSE, FALSE);
941  total += nread;
942  }
943  }
944  ps_end_utt(ps);
945  return total;
946 }

1147 ps_end_utt(ps_decoder_t *ps)
1148 {
1149  int rv, i;
1150 
1151  if (ps->acmod->state == ACMOD_ENDED || ps->acmod->state == ACMOD_IDLE) {
1152  E_ERROR("Utterance is not started\n");
1153  return -1;
1154  }
1155  acmod_end_utt(ps->acmod);
1156 
1157  /* Search any remaining frames. */
1158  if ((rv = ps_search_forward(ps)) < 0) {
1159  ptmr_stop(&ps->perf);
1160  return rv;
1161  }
1162  /* Finish phone loop search. */
1163  if (ps->phone_loop) {
1164  if ((rv = ps_search_finish(ps->phone_loop)) < 0) {
1165  ptmr_stop(&ps->perf);
1166  return rv;
1167  }
1168  }
1169  /* Search any frames remaining in the lookahead window. */
1170  if (ps->acmod->output_frame >= ps->pl_window) {
1171  for (i = ps->acmod->output_frame - ps->pl_window;
1172  i < ps->acmod->output_frame; ++i)
1173  ps_search_step(ps->search, i);
1174  }
1175  /* Finish main search. */
1176  if ((rv = ps_search_finish(ps->search)) < 0) {
1177  ptmr_stop(&ps->perf);
1178  return rv;
1179  }
1180  ptmr_stop(&ps->perf);
1181 
1182  /* Log a backtrace if requested. */
1183  if (cmd_ln_boolean_r(ps->config, "-backtrace")) {
1184  const char* hyp;
1185  ps_seg_t *seg;
1186  int32 score;
1187 
1188  hyp = ps_get_hyp(ps, &score);
1189 
1190  if (hyp != NULL) {
1191  E_INFO("%s (%d)\n", hyp, score);
1192  E_INFO_NOFN("%-20s %-5s %-5s %-5s %-10s %-10s %-3s\n",
1193  "word", "start", "end", "pprob", "ascr", "lscr", "lback");

1204  word, sf, ef, logmath_exp(ps_get_logmath(ps), post),
