114 int
115 main(int argc, char *argv[])
116 {
117  cmd_ln_t *config;
118  ngram_model_t *lm = NULL;
119  logmath_t *lmath;
120  int itype, otype;
121  char const *kase;
122 
123  if ((config = cmd_ln_parse_r(NULL, defn, argc, argv, TRUE)) == NULL)
124  return 1;
125 
126  if (cmd_ln_boolean_r(config, "-help")) {
127  usagemsg(argv[0]);
128  }
129 
130  err_set_debug_level(cmd_ln_int32_r(config, "-debug"));
131 
132  /* Create log math object. */
133  if ((lmath = logmath_init
134  (cmd_ln_float64_r(config, "-logbase"), 0, 0)) == NULL) {
135  E_FATAL("Failed to initialize log math\n");
136  }
137 
138  if (cmd_ln_str_r(config, "-i") == NULL || cmd_ln_str_r(config, "-i") == NULL) {
139  E_ERROR("Please specify both input and output models\n");
140  goto error_out;
141  }
142 
143 
144  /* Load the input language model. */
145  if (cmd_ln_str_r(config, "-ifmt")) {
146  if ((itype = ngram_str_to_type(cmd_ln_str_r(config, "-ifmt")))
147  == NGRAM_INVALID) {
148  E_ERROR("Invalid input type %s\n", cmd_ln_str_r(config, "-ifmt"));
149  goto error_out;
150  }
151  lm = ngram_model_read(config, cmd_ln_str_r(config, "-i"),
152  itype, lmath);
153  }
154  else {
155  lm = ngram_model_read(config, cmd_ln_str_r(config, "-i"),
156  NGRAM_AUTO, lmath);
157  }
158 
159  if (lm == NULL) {
160  E_FATAL("Failed to read the model from the file '%s'\n", cmd_ln_str_r(config, "-i"));
161  }
162 
163  /* Guess or set the output language model type. */
164  if (cmd_ln_str_r(config, "-ofmt")) {
165  if ((otype = ngram_str_to_type(cmd_ln_str_r(config, "-ofmt")))
166  == NGRAM_INVALID) {
167  E_ERROR("Invalid output type %s\n", cmd_ln_str_r(config, "-ofmt"));
168  goto error_out;
169  }
170  }
171  else {
172  otype = ngram_file_name_to_type(cmd_ln_str_r(config, "-o"));
173  }
174 
175  /* Case fold if requested. */
176  if ((kase = cmd_ln_str_r(config, "-case"))) {
177  if (0 == strcmp(kase, "lower")) {
178  ngram_model_casefold(lm, NGRAM_LOWER);
179  }
180  else if (0 == strcmp(kase, "upper")) {
181  ngram_model_casefold(lm, NGRAM_UPPER);
182  }
183  else {
184  E_ERROR("Unknown value for -case: %s\n", kase);
185  goto error_out;
186  }
187  }
188 
189  /* Write the output language model. */
190  if (ngram_model_write(lm, cmd_ln_str_r(config, "-o"), otype) != 0) {
191  E_ERROR("Failed to write language model in format %s to %s\n",
192  ngram_type_to_str(otype), cmd_ln_str_r(config, "-o"));
193  goto error_out;
194  }
195 
196  /* That's all folks! */
197  ngram_model_free(lm);
198  if (lmath) {
199  logmath_free(lmath);
200  }
201  if (config) {
202  cmd_ln_free_r(config);
203  }
204  return 0;
205 
206 error_out:
207  ngram_model_free(lm);
208  if (lmath) {
209  logmath_free(lmath);
210  }
211  if (config) {
212  cmd_ln_free_r(config);
213  }
214  return 1;
215 }
