130 cmn_t *
131 cmn_init(int32 veclen)
132 {
133  cmn_t *cmn;
134  cmn = (cmn_t *) ckd_calloc(1, sizeof(cmn_t));
135  cmn->veclen = veclen;
136  cmn->cmn_mean = (mfcc_t *) ckd_calloc(veclen, sizeof(mfcc_t));
137  cmn->cmn_var = (mfcc_t *) ckd_calloc(veclen, sizeof(mfcc_t));
138  cmn->sum = (mfcc_t *) ckd_calloc(veclen, sizeof(mfcc_t));
139  /* A front-end dependent magic number */
140  cmn->cmn_mean[0] = FLOAT2MFCC(12.0);
141  cmn->nframe = 0;
142  E_INFO("mean[0]= %.2f, mean[1..%d]= 0.0\n",
143  MFCC2FLOAT(cmn->cmn_mean[0]), veclen - 1);
144 
145  return cmn;
146 }

149 void
150 cmn(cmn_t *cmn, mfcc_t ** mfc, int32 varnorm, int32 n_frame)
151 {
152  mfcc_t *mfcp;
153  mfcc_t t;
154  int32 i, f;
155  int32 n_pos_frame;
156 
157  assert(mfc != NULL);
158 
159  if (n_frame <= 0)
160  return;
161 
162  /* If cmn->cmn_mean wasn't NULL, we need to zero the contents */
163  memset(cmn->cmn_mean, 0, cmn->veclen * sizeof(mfcc_t));
164 
165  /* Find mean cep vector for this utterance */
166  for (f = 0, n_pos_frame = 0; f < n_frame; f++) {
167  mfcp = mfc[f];
168 
169  /* Skip zero energy frames */
170  if (mfcp[0] < 0)
171  continue;
172 
173  for (i = 0; i < cmn->veclen; i++) {
174  cmn->cmn_mean[i] += mfcp[i];
175  }
176 
177  n_pos_frame++;
178  }
179 
180  for (i = 0; i < cmn->veclen; i++)
181  cmn->cmn_mean[i] /= n_pos_frame;
182 
183  E_INFO("CMN: ");
184  for (i = 0; i < cmn->veclen; i++)
185  E_INFOCONT("%5.2f ", MFCC2FLOAT(cmn->cmn_mean[i]));
186  E_INFOCONT("\n");
187  if (!varnorm) {
188  /* Subtract mean from each cep vector */
189  for (f = 0; f < n_frame; f++) {
190  mfcp = mfc[f];
191  for (i = 0; i < cmn->veclen; i++)
192  mfcp[i] -= cmn->cmn_mean[i];
193  }
194  }
195  else {
196  /* Scale cep vectors to have unit variance along each dimension, and subtract means */
197  /* If cmn->cmn_var wasn't NULL, we need to zero the contents */
198  memset(cmn->cmn_var, 0, cmn->veclen * sizeof(mfcc_t));
199 
200  for (f = 0; f < n_frame; f++) {
201  mfcp = mfc[f];
202 
203  for (i = 0; i < cmn->veclen; i++) {
204  t = mfcp[i] - cmn->cmn_mean[i];
205  cmn->cmn_var[i] += MFCCMUL(t, t);
206  }
207  }
208  for (i = 0; i < cmn->veclen; i++)
209  /* Inverse Std. Dev, RAH added type case from sqrt */
210  cmn->cmn_var[i] = FLOAT2MFCC(sqrt((float64)n_frame / MFCC2FLOAT(cmn->cmn_var[i])));
211 
212  for (f = 0; f < n_frame; f++) {
213  mfcp = mfc[f];
214  for (i = 0; i < cmn->veclen; i++)
215  mfcp[i] = MFCCMUL((mfcp[i] - cmn->cmn_mean[i]), cmn->cmn_var[i]);
216  }
217  }
218 }
