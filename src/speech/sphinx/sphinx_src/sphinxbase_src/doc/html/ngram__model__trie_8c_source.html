52 /*
53  * Read and return #unigrams, #bigrams, #trigrams as stated in input file.
54  */
55 static int
56 read_counts_arpa(lineiter_t ** li, uint32 * counts, int *order)
57 {
58  int32 ngram, prev_ngram;
59  uint32 ngram_cnt;
60 
61  /* skip file until past the '\data\' marker */
62  while (*li) {
63  if (strcmp((*li)->buf, "\\data\\") == 0)
64  break;
65  *li = lineiter_next(*li);
66  }
67 
68  if (*li == NULL || strcmp((*li)->buf, "\\data\\") != 0) {
69  E_INFO("No \\data\\ mark in LM file\n");
70  return -1;
71  }
72 
73  prev_ngram = 0;
74  *order = 0;
75  while ((*li = lineiter_next(*li))) {
76  if (sscanf((*li)->buf, "ngram %d=%d", &ngram, &ngram_cnt) != 2)
77  break;
78  if (ngram != prev_ngram + 1) {
79  E_ERROR
80  ("Ngram counts in LM file is not in order. %d goes after %d\n",
81  ngram, prev_ngram);
82  return -1;
83  }
84  prev_ngram = ngram;
85  counts[*order] = ngram_cnt;
86  (*order)++;
87  }
88 
89  if (*li == NULL) {
90  E_ERROR("EOF while reading ngram counts\n");
91  return -1;
92  }
93 
94  return 0;
95 }

97 static int
98 read_1grams_arpa(lineiter_t ** li, uint32 count, ngram_model_t * base,
99  unigram_t * unigrams)
100 {
101  uint32 i;
102  int n;
103  int n_parts;
104  char *wptr[3];
105 
106  while (*li && strcmp((*li)->buf, "\\1-grams:") != 0) {
107  *li = lineiter_next(*li);
108  }
109  if (*li == NULL) {
110  E_ERROR_SYSTEM("Failed to read \\1-grams: mark");
111  return -1;
112  }
113 
114  n_parts = 2;
115  for (i = 0; i < count; i++) {
116  unigram_t *unigram;
117 
118  *li = lineiter_next(*li);
119  if (*li == NULL) {
120  E_ERROR
121  ("Unexpected end of ARPA file. Failed to read %dth unigram\n",
122  i + 1);
123  return -1;
124  }
125  if ((n = str2words((*li)->buf, wptr, 3)) < n_parts) {
126  E_ERROR("Format error at line %s, Failed to read unigrams\n", (*li)->buf);
127  return -1;
128  }
129 
130  unigram = &unigrams[i];
131  unigram->prob =
132  logmath_log10_to_log_float(base->lmath, atof_c(wptr[0]));
133  if (unigram->prob > 0) {
134  E_WARN("Unigram '%s' has positive probability\n", wptr[1]);
135  unigram->prob = 0;
136  }
137  if (n == n_parts + 1) {
138  unigram->bo =
139  logmath_log10_to_log_float(base->lmath,
140  atof_c(wptr[2]));
141  }
142  else {
143  unigram->bo = 0.0f;
144  }
145 
146  /* TODO: classify float with fpclassify and warn if bad value occurred */
147  base->word_str[i] = ckd_salloc(wptr[1]);
148  }
149 
150  /* fill hash-table that maps unigram names to their word ids */
151  for (i = 0; i < count; i++) {
152  if ((hash_table_enter
153  (base->wid, base->word_str[i],
154  (void *) (long) i)) != (void *) (long) i) {
155  E_WARN("Duplicate word in dictionary: %s\n",
156  base->word_str[i]);
157  }
158  }
159  return 0;
160 }

162 ngram_model_t *
163 ngram_model_trie_read_arpa(cmd_ln_t * config,
164  const char *path, logmath_t * lmath)
165 {
166  FILE *fp;
167  lineiter_t *li;
168  ngram_model_trie_t *model;
169  ngram_model_t *base;
170  ngram_raw_t **raw_ngrams;
171  int32 is_pipe;
172  uint32 counts[NGRAM_MAX_ORDER];
173  int order;
174  int i;
175 
176  E_INFO("Trying to read LM in arpa format\n");
177  if ((fp = fopen_comp(path, "r", &is_pipe)) == NULL) {
178  E_ERROR("File %s not found\n", path);
179  return NULL;
180  }
181 
182  model = (ngram_model_trie_t *) ckd_calloc(1, sizeof(*model));
183  li = lineiter_start_clean(fp);
184  /* Read n-gram counts from file */
185  if (read_counts_arpa(&li, counts, &order) == -1) {
186  ckd_free(model);
187  lineiter_free(li);
188  fclose_comp(fp, is_pipe);
189  return NULL;
190  }
191 
192  E_INFO("LM of order %d\n", order);
193  for (i = 0; i < order; i++) {
194  E_INFO("#%d-grams: %d\n", i + 1, counts[i]);
195  }
196 
197  base = &model->base;
198  ngram_model_init(base, &ngram_model_trie_funcs, lmath, order,
199  (int32) counts[0]);
200  base->writable = TRUE;
201 
202  model->trie = lm_trie_create(counts[0], order);
203  if (read_1grams_arpa(&li, counts[0], base, model->trie->unigrams) < 0) {
204  ngram_model_free(base);
205  lineiter_free(li);
206  fclose_comp(fp, is_pipe);
207  return NULL;
208  }
209 
210  if (order > 1) {
211  raw_ngrams =
212  ngrams_raw_read_arpa(&li, base->lmath, counts, order,
213  base->wid);
214  lm_trie_build(model->trie, raw_ngrams, counts, base->n_counts, order);
215  ngrams_raw_free(raw_ngrams, counts, order);
216  }
217 
218  lineiter_free(li);
219  fclose_comp(fp, is_pipe);
220 
221  return base;
222 }

224 int
225 ngram_model_trie_write_arpa(ngram_model_t * base, const char *path)
226 {
227  int i;
228  uint32 j;
229  ngram_model_trie_t *model = (ngram_model_trie_t *) base;
230  FILE *fp = fopen(path, "w");
231  if (!fp) {
232  E_ERROR("Unable to open %s to write arpa LM from trie\n", path);
233  return -1;
234  }
235  fprintf(fp,
236  "This is an ARPA-format language model file, generated by CMU Sphinx\n");
237  /* Write N-gram counts. */
238  fprintf(fp, "\\data\\\n");
239  for (i = 0; i < base->n; ++i) {
240  fprintf(fp, "ngram %d=%d\n", i + 1, base->n_counts[i]);
241  }
242  /* Write 1-grams */
243  fprintf(fp, "\n\\1-grams:\n");
244  for (j = 0; j < base->n_counts[0]; j++) {
245  unigram_t *unigram = &model->trie->unigrams[j];
246  fprintf(fp, "%.4f\t%s",
247  logmath_log_float_to_log10(base->lmath, unigram->prob),
248  base->word_str[j]);
249  if (base->n > 1) {
250  fprintf(fp, "\t%.4f",
251  logmath_log_float_to_log10(base->lmath, unigram->bo));
252  }
253  fprintf(fp, "\n");
254  }
255  /* Write ngrams */
256  if (base->n > 1) {
257  for (i = 2; i <= base->n; ++i) {
258  ngram_raw_t *raw_ngrams =
259  (ngram_raw_t *) ckd_calloc((size_t) base->n_counts[i - 1],
260  sizeof(*raw_ngrams));
261  uint32 raw_ngram_idx;
262  uint32 j;
263  uint32 hist[NGRAM_MAX_ORDER];
264  node_range_t range;
265  raw_ngram_idx = 0;
266  range.begin = range.end = 0;
267 
268  /* we need to iterate over a trie here. recursion should do the job */
269  lm_trie_fill_raw_ngram(model->trie, raw_ngrams,
270  &raw_ngram_idx, base->n_counts, range, hist, 0,
271  i, base->n);
272  assert(raw_ngram_idx == base->n_counts[i - 1]);
273  qsort(raw_ngrams, (size_t) base->n_counts[i - 1],
274  sizeof(ngram_raw_t), &ngram_ord_comparator);
275 
276  fprintf(fp, "\n\\%d-grams:\n", i);
277  for (j = 0; j < base->n_counts[i - 1]; j++) {
278  int k;
279  fprintf(fp, "%.4f", logmath_log_float_to_log10(base->lmath, raw_ngrams[j].prob));
280  for (k = 0; k < i; k++) {
281  fprintf(fp, "\t%s",
282  base->word_str[raw_ngrams[j].words[k]]);
283  }
284  ckd_free(raw_ngrams[j].words);
285  if (i < base->n) {
286  fprintf(fp, "\t%.4f", logmath_log_float_to_log10(base->lmath, raw_ngrams[j].backoff));
287  }
288  fprintf(fp, "\n");
289  }
290  ckd_free(raw_ngrams);
291  }
292  }
293  fprintf(fp, "\n\\end\\\n");
294  return fclose(fp);
295 }

297 static void
298 read_word_str(ngram_model_t * base, FILE * fp)
299 {
300  int32 k;
301  uint32 i, j;
302  char *tmp_word_str;
303  /* read ascii word strings */
304  base->writable = TRUE;
305  fread(&k, sizeof(k), 1, fp);
306  tmp_word_str = (char *) ckd_calloc((size_t) k, 1);
307  fread(tmp_word_str, 1, (size_t) k, fp);
308 
309  /* First make sure string just read contains n_counts[0] words (PARANOIA!!) */
310  for (i = 0, j = 0; i < (uint32) k; i++)
311  if (tmp_word_str[i] == '\0')
312  j++;
313  if (j != base->n_counts[0]) {
314  E_ERROR
315  ("Error reading word strings (%d doesn't match n_unigrams %d)\n",
316  j, base->n_counts[0]);
317  }
318 
319  /* Break up string just read into words */
320  j = 0;
321  for (i = 0; i < base->n_counts[0]; i++) {
322  base->word_str[i] = ckd_salloc(tmp_word_str + j);
323  if (hash_table_enter(base->wid, base->word_str[i],
324  (void *) (long) i) != (void *) (long) i) {
325  E_WARN("Duplicate word in dictionary: %s\n",
326  base->word_str[i]);
327  }
328  j += strlen(base->word_str[i]) + 1;
329  }
330  free(tmp_word_str);
331 }

333 ngram_model_t *
334 ngram_model_trie_read_bin(cmd_ln_t * config,
335  const char *path, logmath_t * lmath)
336 {
337  int32 is_pipe;
338  FILE *fp;
339  size_t hdr_size;
340  char *hdr;
341  int cmp_res;
342  uint8 i, order;
343  uint32 counts[NGRAM_MAX_ORDER];
344  ngram_model_trie_t *model;
345  ngram_model_t *base;
346 
347  E_INFO("Trying to read LM in trie binary format\n");
348  if ((fp = fopen_comp(path, "rb", &is_pipe)) == NULL) {
349  E_ERROR("File %s not found\n", path);
350  return NULL;
351  }
352  hdr_size = strlen(trie_hdr);
353  hdr = (char *) ckd_calloc(hdr_size + 1, sizeof(*hdr));
354  fread(hdr, sizeof(*hdr), hdr_size, fp);
355  cmp_res = strcmp(hdr, trie_hdr);
356  ckd_free(hdr);
357  if (cmp_res) {
358  E_INFO("Header doesn't match\n");
359  fclose_comp(fp, is_pipe);
360  return NULL;
361  }
362  model = (ngram_model_trie_t *) ckd_calloc(1, sizeof(*model));
363  base = &model->base;
364  fread(&order, sizeof(order), 1, fp);
365  for (i = 0; i < order; i++) {
366  fread(&counts[i], sizeof(counts[i]), 1, fp);
367  }
368  ngram_model_init(base, &ngram_model_trie_funcs, lmath, order,
369  (int32) counts[0]);
370  for (i = 0; i < order; i++) {
371  base->n_counts[i] = counts[i];
372  }
373 
374  model->trie = lm_trie_read_bin(counts, order, fp);
375  read_word_str(base, fp);
376  fclose_comp(fp, is_pipe);
377 
378  return base;
379 }

381 static void
382 write_word_str(FILE * fp, ngram_model_t * model)
383 {
384  int32 k;
385  uint32 i;
386 
387  k = 0;
388  for (i = 0; i < model->n_counts[0]; i++)
389  k += strlen(model->word_str[i]) + 1;
390  fwrite(&k, sizeof(k), 1, fp);
391  for (i = 0; i < model->n_counts[0]; i++)
392  fwrite(model->word_str[i], 1, strlen(model->word_str[i]) + 1, fp);
393 }

395 int
396 ngram_model_trie_write_bin(ngram_model_t * base, const char *path)
397 {
398  int i;
399  int32 is_pipe;
400  ngram_model_trie_t *model = (ngram_model_trie_t *) base;
401  FILE *fp = fopen_comp(path, "wb", &is_pipe);
402  if (!fp) {
403  E_ERROR("Unable to open %s to write binary trie LM\n", path);
404  return -1;
405  }
406 
407  fwrite(trie_hdr, sizeof(*trie_hdr), strlen(trie_hdr), fp);
408  fwrite(&model->base.n, sizeof(model->base.n), 1, fp);
409  for (i = 0; i < model->base.n; i++) {
410  fwrite(&model->base.n_counts[i], sizeof(model->base.n_counts[i]),
411  1, fp);
412  }
413  lm_trie_write_bin(model->trie, base->n_counts[0], fp);
414  write_word_str(fp, base);
415  fclose_comp(fp, is_pipe);
416  return 0;
417 }

419 ngram_model_t *
420 ngram_model_trie_read_dmp(cmd_ln_t * config,
421  const char *file_name, logmath_t * lmath)
422 {
423  uint8 do_swap;
424  int32 is_pipe;
425  int32 k;
426  uint32 j;
427  int32 vn, ts;
428  int32 count;
429  uint32 counts[3];
430  uint32 *unigram_next;
431  int order;
432  char str[1024];
433  FILE *fp;
434  ngram_model_trie_t *model;
435  ngram_model_t *base;
436  ngram_raw_t **raw_ngrams;
437 
438  E_INFO("Trying to read LM in DMP format\n");
439  if ((fp = fopen_comp(file_name, "rb", &is_pipe)) == NULL) {
440  E_ERROR("Dump file %s not found\n", file_name);
441  return NULL;
442  }
443 
444  do_swap = FALSE;
445  fread(&k, sizeof(k), 1, fp);
446  if (k != strlen(dmp_hdr) + 1) {
447  SWAP_INT32(&k);
448  if (k != strlen(dmp_hdr) + 1) {
449  E_ERROR
450  ("Wrong magic header size number %x: %s is not a dump file\n",
451  k, file_name);
452  return NULL;
453  }
454  do_swap = 1;
455  }
456  if (fread(str, 1, k, fp) != (size_t) k) {
457  E_ERROR("Cannot read header\n");
458  return NULL;
459  }
460  if (strncmp(str, dmp_hdr, k) != 0) {
461  E_ERROR("Wrong header %s: %s is not a dump file\n", dmp_hdr);
462  return NULL;
463  }
464 
465  if (fread(&k, sizeof(k), 1, fp) != 1)
466  return NULL;
467  if (do_swap)
468  SWAP_INT32(&k);
469  if (fread(str, 1, k, fp) != (size_t) k) {
470  E_ERROR("Cannot read LM filename in header\n");
471  return NULL;
472  }
473 
474  /* read version#, if present (must be <= 0) */
475  if (fread(&vn, sizeof(vn), 1, fp) != 1)
476  return NULL;
477  if (do_swap)
478  SWAP_INT32(&vn);
479  if (vn <= 0) {
480  /* read and don't compare timestamps (we don't care) */
481  if (fread(&ts, sizeof(ts), 1, fp) != 1)
482  return NULL;
483  if (do_swap)
484  SWAP_INT32(&ts);
485 
486  /* read and skip format description */
487  for (;;) {
488  if (fread(&k, sizeof(k), 1, fp) != 1)
489  return NULL;
490  if (do_swap)
491  SWAP_INT32(&k);
492  if (k == 0)
493  break;
494  if (fread(str, 1, k, fp) != (size_t) k) {
495  E_ERROR("Failed to read word\n");
496  return NULL;
497  }
498  }
499  /* read model->ucount */
500  if (fread(&count, sizeof(count), 1, fp) != 1)
501  return NULL;
502  if (do_swap)
503  SWAP_INT32(&count);
504  counts[0] = count;
505  }
506  else {
507  counts[0] = vn;
508  }
509  /* read model->bcount, tcount */
510  if (fread(&count, sizeof(count), 1, fp) != 1)
511  return NULL;
512  if (do_swap)
513  SWAP_INT32(&count);
514  counts[1] = count;
515  if (fread(&count, sizeof(count), 1, fp) != 1)
516  return NULL;
517  if (do_swap)
518  SWAP_INT32(&count);
519  counts[2] = count;
520  E_INFO("ngrams 1=%d, 2=%d, 3=%d\n", counts[0], counts[1], counts[2]);
521 
522  model = (ngram_model_trie_t *) ckd_calloc(1, sizeof(*model));
523  base = &model->base;
524  if (counts[2] > 0)
525  order = 3;
526  else if (counts[1] > 0)
527  order = 2;
528  else
529  order = 1;
530  ngram_model_init(base, &ngram_model_trie_funcs, lmath, order,
531  (int32) counts[0]);
532 
533  model->trie = lm_trie_create(counts[0], order);
534 
535  unigram_next =
536  (uint32 *) ckd_calloc((int32) counts[0] + 1, sizeof(unigram_next));
537  for (j = 0; j <= (int32) counts[0]; j++) {
538  int32 bigrams;
539  int32 mapid;
540  dmp_weight_t weightp;
541  dmp_weight_t weightb;
542 
543  /* Skip over the mapping ID, we don't care about it. */
544  /* Read the weights from actual unigram structure. */
545  fread(&mapid, sizeof(int32), 1, fp);
546  fread(&weightp, sizeof(weightp), 1, fp);
547  fread(&weightb, sizeof(weightb), 1, fp);
548  fread(&bigrams, sizeof(int32), 1, fp);
549  if (do_swap) {
550  SWAP_INT32(&weightp.l);
551  SWAP_INT32(&weightb.l);
552  SWAP_INT32(&bigrams);
553  }
554  model->trie->unigrams[j].prob = logmath_log10_to_log_float(lmath, weightp.f);
555  model->trie->unigrams[j].bo = logmath_log10_to_log_float(lmath, weightb.f);
556  model->trie->unigrams[j].next = bigrams;
557  unigram_next[j] = bigrams;
558  }
559 
560  if (order > 1) {
561  raw_ngrams =
562  ngrams_raw_read_dmp(fp, lmath, counts, order, unigram_next,
563  do_swap);
564  lm_trie_build(model->trie, raw_ngrams, counts, base->n_counts, order);
565  ngrams_raw_free(raw_ngrams, counts, order);
566  }
567 
568  /* Sentinel unigram and bigrams read before */
569  ckd_free(unigram_next);
570 
571  /* read ascii word strings */
572  read_word_str(base, fp);
573 
574  fclose_comp(fp, is_pipe);
575  return base;
576 }

600 static int32
601 ngram_model_trie_raw_score(ngram_model_t * base, int32 wid, int32 * hist,
602  int32 n_hist, int32 * n_used)
603 {
604  int32 i;
605  ngram_model_trie_t *model = (ngram_model_trie_t *) base;
606 
607  if (n_hist > model->base.n - 1)
608  n_hist = model->base.n - 1;
609  for (i = 0; i < n_hist; i++) {
610  if (hist[i] < 0) {
611  n_hist = i;
612  break;
613  }
614  }
615 
616  return (int32) lm_trie_score(model->trie, model->base.n, wid, hist,
617  n_hist, n_used);
618 }

629 static int32
630 lm_trie_add_ug(ngram_model_t * base, int32 wid, int32 lweight)
631 {
632  ngram_model_trie_t *model = (ngram_model_trie_t *) base;
633 
634  /* This would be very bad if this happened! */
635  assert(!NGRAM_IS_CLASSWID(wid));
636 
637  /* Reallocate unigram array. */
638  model->trie->unigrams =
639  (unigram_t *) ckd_realloc(model->trie->unigrams,
640  sizeof(*model->trie->unigrams) *
641  (base->n_1g_alloc + 1));
642  memset(model->trie->unigrams + (base->n_counts[0] + 1), 0,
643  (size_t) (base->n_1g_alloc -
644  base->n_counts[0]) * sizeof(*model->trie->unigrams));
645  ++base->n_counts[0];
646  lweight += logmath_log(base->lmath, 1.0 / base->n_counts[0]);
647  model->trie->unigrams[wid + 1].next = model->trie->unigrams[wid].next;
648  model->trie->unigrams[wid].prob = (float) lweight;
649  /* This unigram by definition doesn't participate in any bigrams,
650  * so its backoff weight is undefined and next pointer same as in finish unigram*/
651  model->trie->unigrams[wid].bo = 0;
652  /* Finally, increase the unigram count */
653  /* FIXME: Note that this can actually be quite bogus due to the
654  * presence of class words. If wid falls outside the unigram
655  * count, increase it to compensate, at the cost of no longer
656  * really knowing how many unigrams we have :( */
657  if ((uint32) wid >= base->n_counts[0])
658  base->n_counts[0] = wid + 1;
659 
660  return (int32) weight_score(base, lweight);
661 }
