79 #ifndef FIXED_POINT
80 static void
81 cmn_diff(int16 const *signal, float *out_diff, int ndiff)
82 {
83  double cum;
84  int t, j;
85 
86  cum = 0.0f;
87  out_diff[0] = 1.0f;
88 
89  for (t = 1; t < ndiff; ++t) {
90  float dd;
91  dd = 0.0f;
92  for (j = 0; j < ndiff; ++j) {
93  int diff = signal[j] - signal[t + j];
94  dd += (diff * diff);
95  }
96  cum += dd;
97  out_diff[t] = (float)(dd * t / cum);
98  }
99 }
100 #else
101 static void
102 cmn_diff(int16 const *signal, int32 *out_diff, int ndiff)
103 {
104  uint32 cum, cshift;
105  int32 t, tscale;
106 
107  out_diff[0] = 32768;
108  cum = 0;
109  cshift = 0;
110 
111  /* Determine how many bits we can scale t up by below. */
112  for (tscale = 0; tscale < 32; ++tscale)
113  if (ndiff & (1<<(31-tscale)))
114  break;
115  --tscale; /* Avoid teh overflowz. */
116  /* printf("tscale is %d (ndiff - 1) << tscale is %d\n",
117  tscale, (ndiff-1) << tscale); */
118 
119  /* Somewhat elaborate block floating point implementation.
120  * The fp implementation of this is really a lot simpler. */
121  for (t = 1; t < ndiff; ++t) {
122  uint32 dd, dshift, norm;
123  int j;
124 
125  dd = 0;
126  dshift = 0;
127  for (j = 0; j < ndiff; ++j) {
128  int diff = signal[j] - signal[t + j];
129  /* Guard against overflows. */
130  if (dd > (1UL<<tscale)) {
131  dd >>= 1;
132  ++dshift;
133  }
134  dd += (diff * diff) >> dshift;
135  }
136  /* Make sure the diffs and cum are shifted to the same
137  * scaling factor (usually dshift will be zero) */
138  if (dshift > cshift) {
139  cum += dd << (dshift-cshift);
140  }
141  else {
142  cum += dd >> (cshift-dshift);
143  }
144 
145  /* Guard against overflows and also ensure that (t<<tscale) > cum. */
146  while (cum > (1UL<<tscale)) {
147  cum >>= 1;
148  ++cshift;
149  }
150  /* Avoid divide-by-zero! */
151  if (cum == 0) cum = 1;
152  /* Calculate the normalizer in high precision. */
153  norm = (t << tscale) / cum;
154  /* Do a long multiply and shift down to Q15. */
155  out_diff[t] = (int32)(((long long)dd * norm)
156  >> (tscale - 15 + cshift - dshift));
157  /* printf("dd %d cshift %d dshift %d scaledt %d cum %d norm %d cmn %d\n",
158  dd, cshift, dshift, (t<<tscale), cum, norm, out_diff[t]); */
159  }
160 }

163 yin_t *
164 yin_init(int frame_size, float search_threshold,
165  float search_range, int smooth_window)
166 {
167  yin_t *pe;
168 
169  pe = ckd_calloc(1, sizeof(*pe));
170  pe->frame_size = frame_size;
171 #ifndef FIXED_POINT
172  pe->search_threshold = search_threshold;
173  pe->search_range = search_range;
174 #else
175  pe->search_threshold = (uint16)(search_threshold * 32768);
176  pe->search_range = (uint16)(search_range * 32768);
177 #endif
178  pe->wsize = smooth_window * 2 + 1;
179  pe->diff_window = ckd_calloc_2d(pe->wsize,
180  pe->frame_size / 2,
181  sizeof(**pe->diff_window));
182  pe->period_window = ckd_calloc(pe->wsize,
183  sizeof(*pe->period_window));
184  pe->frame = ckd_calloc(pe->frame_size, sizeof(*pe->frame));
185  return pe;
186 }

211 int
212 #ifndef FIXED_POINT
213 thresholded_search(float *diff_window, float threshold, int start, int end)
214 #else
215 thresholded_search(int32 *diff_window, fixed32 threshold, int start, int end)
216 #endif
217 {
218  int i, argmin;
219 #ifndef FIXED_POINT
220  float min;
221 #else
222  int min;
223 #endif
224 
225  min = diff_window[start];
226  argmin = start;
227  for (i = start + 1; i < end; ++i) {
228 #ifndef FIXED_POINT
229  float diff = diff_window[i];
230 #else
231  int diff = diff_window[i];
232 #endif
233 
234  if (diff < threshold) {
235  min = diff;
236  argmin = i;
237  break;
238  }
239  if (diff < min) {
240  min = diff;
241  argmin = i;
242  }
243  }
244  return argmin;
245 }

253 void
254 yin_write(yin_t *pe, int16 const *frame)
255 {
256  int outptr, difflen;
257 
258  /* Rotate the window one frame forward. */
259  ++pe->wstart;
260  /* Fill in the frame before wstart. */
261  outptr = pe->wstart - 1;
262  /* Wrap around the window pointer. */
263  if (pe->wstart == pe->wsize)
264  pe->wstart = 0;
265 
266  /* Now calculate normalized difference function. */
267  difflen = pe->frame_size / 2;
268  cmn_diff(frame, pe->diff_window[outptr], difflen);
269 
270  /* Find the first point under threshold. If not found, then
271  * use the absolute minimum. */
272  pe->period_window[outptr]
273  = thresholded_search(pe->diff_window[outptr],
274  pe->search_threshold, 0, difflen);
275 
276  /* Increment total number of frames. */
277  ++pe->nfr;
278 }

286 int
287 yin_read(yin_t *pe, uint16 *out_period, float *out_bestdiff)
288 {
289  int wstart, wlen, half_wsize, i;
290  int best, search_width, low_period, high_period;
291 #ifndef FIXED_POINT
292  float best_diff;
293 #else
294  int best_diff;
295 #endif
296 
297  half_wsize = (pe->wsize-1)/2;
298  /* Without any smoothing, just return the current value (don't
299  * need to do anything to the current poitner either). */
300  if (half_wsize == 0) {
301  if (pe->endut)
302  return 0;
303  *out_period = pe->period_window[0];
304 #ifndef FIXED_POINT
305  *out_bestdiff = pe->diff_window[0][pe->period_window[0]];
306 #else
307  *out_bestdiff = pe->diff_window[0][pe->period_window[0]] / 32768.0f;
308 #endif
309  return 1;
310  }
311 
312  /* We can't do anything unless we have at least (wsize-1)/2 + 1
313  * frames, unless we're at the end of the utterance. */
314  if (pe->endut == 0 && pe->nfr < half_wsize + 1) {
315  /* Don't increment the current pointer either. */
316  return 0;
317  }
318 
319  /* Establish the smoothing window. */
320  /* End of utterance. */
321  if (pe->endut) {
322  /* We are done (no more data) when pe->wcur = pe->wstart. */
323  if (pe->wcur == pe->wstart)
324  return 0;
325  /* I.e. pe->wcur (circular minus) half_wsize. */
326  wstart = (pe->wcur + pe->wsize - half_wsize) % pe->wsize;
327  /* Number of frames from wstart up to pe->wstart. */
328  wlen = pe->wstart - wstart;
329  if (wlen < 0) wlen += pe->wsize;
330  /*printf("ENDUT! ");*/
331  }
332  /* Beginning of utterance. */
333  else if (pe->nfr < pe->wsize) {
334  wstart = 0;
335  wlen = pe->nfr;
336  }
337  /* Normal case, it is what it is. */
338  else {
339  wstart = pe->wstart;
340  wlen = pe->wsize;
341  }
342 
343  /* Now (finally) look for the best local estimate. */
344  /* printf("Searching for local estimate in %d frames around %d\n",
345  wlen, pe->nfr + 1 - wlen); */
346  best = pe->period_window[pe->wcur];
347  best_diff = pe->diff_window[pe->wcur][best];
348  for (i = 0; i < wlen; ++i) {
349  int j = wstart + i;
350 #ifndef FIXED_POINT
351  float diff;
352 #else
353  int diff;
354 #endif
355 
356  j %= pe->wsize;
357  diff = pe->diff_window[j][pe->period_window[j]];
358  /* printf("%.2f,%.2f ", 1.0 - (double)diff/32768,
359  pe->period_window[j] ? 8000.0/pe->period_window[j] : 8000.0); */
360  if (diff < best_diff) {
361  best_diff = diff;
362  best = pe->period_window[j];
363  }
364  }
365  /* printf("best: %.2f, %.2f\n", 1.0 - (double)best_diff/32768,
366  best ? 8000.0/best : 8000.0); */
367  /* If it's the same as the current one then return it. */
368  if (best == pe->period_window[pe->wcur]) {
369  /* Increment the current pointer. */
370  if (++pe->wcur == pe->wsize)
371  pe->wcur = 0;
372  *out_period = best;
373 #ifndef FIXED_POINT
374  *out_bestdiff = best_diff;
375 #else
376  *out_bestdiff = best_diff / 32768.0f;
377 #endif
378  return 1;
379  }
380  /* Otherwise, redo the search inside a narrower range. */
381 #ifndef FIXED_POINT
382  search_width = (int)(best * pe->search_range);
383 #else
384  search_width = best * pe->search_range / 32768;
385 #endif
386  /* printf("Search width = %d * %.2f = %d\n",
387  best, (double)pe->search_range/32768, search_width); */
388  if (search_width == 0) search_width = 1;
389  low_period = best - search_width;
390  high_period = best + search_width;
391  if (low_period < 0) low_period = 0;
392  if (high_period > pe->frame_size / 2) high_period = pe->frame_size / 2;
393  /* printf("Searching from %d to %d\n", low_period, high_period); */
394  best = thresholded_search(pe->diff_window[pe->wcur],
395  pe->search_threshold,
396  low_period, high_period);
397  best_diff = pe->diff_window[pe->wcur][best];
398 
399  if (out_period)
400  *out_period = (best > 65535) ? 65535 : best;
401  if (out_bestdiff) {
402 #ifndef FIXED_POINT
403  *out_bestdiff = (best_diff > 1.0f) ? 1.0f : best_diff;
404 #else
405  *out_bestdiff = (best_diff > 32768) ? 1.0f : best_diff / 32768.0f;
406 #endif
407  }
408 
409  /* Increment the current pointer. */
410  if (++pe->wcur == pe->wsize)
411  pe->wcur = 0;
412  return 1;
413 }
